Future<List<SimpleLog>> getSimpleLogs() async {
  final dir = await getApplicationSupportDirectory();
  final rawLogs =
      await Isolate.run(() => IsolateLogWriter.readLogs(dir.path, 'simple'));
  return rawLogs.map((e) => SimpleLog.fromJson(e)).toList();
}
